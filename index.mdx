```javascript rest
// args to array
const a = [1, 2, 3];

const func_prev = (one, two, three) => {
    console.log(`inputs are ${one}, ${two} and ${three}`);
};

const func = (...args) => {
    console.log(args);
};

func_prev(...a);
func(...a);
```

export default Code;

```javascript destructuring
// array assignments
const a = [1, 2, 3, 4, 5];
const [x, y] = a;
const [x_1, ...rest] = a;
console.log(x, y);
console.log(x_1, test);

// swap
const b = 1;
const c = 2;
[b, c] = [c, b];
console.log(b);
console.log(c);

// obj destructuring
const props = { id: 1, name: "foo" };
const { name } = props;
console.log(name);
```
